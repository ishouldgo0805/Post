abstract class Attachments {
    abstract var type: String
}

abstract class PhotoAttachment() : Attachments() {
    override var type: String = "Photo Attachment"
    open val id: Int = 0

}


class Photo(
    override var type: String = "Photo",
    override val id: Int = 1,
    val albumId: Int = 2,
    val ownerId: Int = 3,
    val userId: Int = 4,
    val text: String = "",
    val date: Int = 1,
    val sizes: Int = 1,
    val width: Int = 1,
    val height: Int = 1
) : PhotoAttachment()
